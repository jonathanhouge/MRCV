for getting reviewers:
python3 -m listscraper {URL} -r True -on "{movie}-reviewers.json" -p #~#
ex: python3 -m listscraper https://letterboxd.com/film/pacific-rim/reviews/ -r True -on "pacific-rim-reviewers.json" -p 1

for getting films:
python3 -m listscraper {URL} -p #~# -on "{year}-{category}"
ex: python3 -m listscraper https://letterboxd.com/films/year/2023/genre/documentary/by/shortest/ -p 8~45 -on "2023-docs-short"
