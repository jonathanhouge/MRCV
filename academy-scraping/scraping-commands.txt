for getting watchers:
python3 -m listscraper {URL} -w True -on "{movie}-watchers.json" -p #~#
ex: python3 -m listscraper https://letterboxd.com/film/terminator-2-judgment-day/members/ -w True -on "t2-reviewers.json" -p 1

for getting films:
python3 -m listscraper {URL} -p #~# -on "{year}-{category}"
ex: python3 -m listscraper https://letterboxd.com/films/year/2023/genre/documentary/by/shortest/ -p 8~45 -on "2023-docs-short"
