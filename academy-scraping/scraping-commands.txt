for getting ballots:
python3 -m listscraper null -c {category-scrape-path} -cat {member-filters} -on "{category}-ballots.json"
ex: python3 -m listscraper null -c "academy-scraping/animated-short/2013-animated-short.json" -cat "2013/genre/animation/by/shortest/" -on "2013-animated-short-ballots.json"

for getting films:
python3 -m listscraper {URL} -p #~# -on "{year}-{category}"
ex: python3 -m listscraper https://letterboxd.com/films/year/2023/genre/documentary/by/shortest/ -p 8~45 -on "2023-docs-short"
